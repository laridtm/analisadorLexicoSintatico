<identificador> ::=  <palavra> (<palavra> | <digito> | "_")*

<palavra_reservada> ::= "var" | "se" | "senao" | "enquanto" | "classe"

<operador> ::= "+" | "*" | "-" | "/" | "<" | ">" | "<=" | ">=" | "==" | "!="

<delimitador> ::= ";"
